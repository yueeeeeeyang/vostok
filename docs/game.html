🎮 Vostok.Game

Game 游戏引擎模块

实时多人游戏服务端引擎，基于房间 Tick 驱动模型，提供分片线程池、热点房间迁移、帧同步、玩家生命周期、断线重连、匹配队列等核心能力。

初始化

Vostok.Game.init(new VKGameConfig()
    .tickRate(20)            // 每秒 20 Tick（50ms/tick）
    .maxRooms(10000)
    .maxPlayersPerRoom(128)
    .tickWorkerThreads(8)
    .antiCheatEnabled(true)
    .matchmakingEnabled(false)
);

注册游戏逻辑

Vostok.Game.registerLogic("chess", new VKGameLogic() {

    public void onTick(VKGameRoom room, long tickSeq) {
        // 每 Tick 调用一次，tickSeq 单调递增
        List<VKGameCommand> cmds = room.pollCommands();
        for (VKGameCommand cmd : cmds) {
            applyMove(room, cmd);
        }
    }

    public void onPlayerJoin(VKGameRoom room, VKGamePlayerSession player) {
        Vostok.Log.info("player {} joined room {}", player.playerId(), room.roomId());
    }

    public void onPlayerLeave(VKGameRoom room, VKGamePlayerSession player) {}

    public void onRoomStart(VKGameRoom room) {}
    public void onRoomEnd(VKGameRoom room) {}
});

房间管理

// 创建房间（自动生成 roomId）
VKGameRoom room = Vostok.Game.createRoom("chess");

// 创建指定 roomId 的房间
VKGameRoom room = Vostok.Game.createRoom("room-42", "chess");

// 帧同步模式
VKGameRoom room = Vostok.Game.createRoom("room-43", "fps", true);

// 查询
VKGameRoom r = Vostok.Game.room("room-42");
Set<String> ids = Vostok.Game.roomIds();

// 移除房间
Vostok.Game.removeRoom("room-42");
Vostok.Game.drainRoom("room-42", "game over");  // 优雅排空

玩家管理

// 玩家加入
Vostok.Game.join("room-42", "player-1");

// 携带加入 Token（服务端签发的认证令牌）
Vostok.Game.joinWithToken("room-42", "player-1", joinToken);

// 玩家离开
Vostok.Game.leave("room-42", "player-1");

// 断线（保留 Session，可重连）
Vostok.Game.disconnect("room-42", "player-1");

// 断线重连
Vostok.Game.reconnect("room-42", "player-1", sessionToken);

发送指令

// 提交玩家指令（完整签名）
Vostok.Game.submit("room-42", VKGameCommand.builder()
    .playerId("player-1")
    .type("MOVE")
    .payload(Map.of("from", "e2", "to", "e4"))
    .build());

// 便捷提交（roomId, playerId, type, payload）
Vostok.Game.submit("room-42", "player-1", "MOVE",
    Map.of("from", "e2", "to", "e4"));

// 非阻塞提交（队列满时返回 false）
boolean accepted = Vostok.Game.trySubmit("room-42", cmd);

消息发布

// 批量发布消息给玩家
Vostok.Game.publishMessages(List.of(
    new VKGameMessagePublishCommand("player-1", "UPDATE", payload1),
    new VKGameMessagePublishCommand("player-2", "UPDATE", payload2)
));

// 拉取玩家消息（poll 模式）
List<VKGameMessage> msgs = Vostok.Game.pollMessages(
    "player-1", VKGameMessageScope.ROOM, "room-42", 0L, 50, null);

// 确认已消费
Vostok.Game.ackMessages("player-1", List.of(msg1.id(), msg2.id()));

匹配系统

// 提交匹配请求，获取 ticketId
String ticketId = Vostok.Game.enqueueMatch(
    VKGameMatchRequest.builder()
        .gameType("chess")
        .playerId("player-1")
        .ratingRange(100)
        .build()
);

// 异步轮询匹配结果（客户端定时查询）
VKGameMatchResult result = Vostok.Game.pollMatchResult(ticketId);
if (result != null && result.matched()) {
    String roomId = result.roomId();
    // 确认已接受匹配
    Vostok.Game.ackMatchFound(ticketId, "player-1");
    Vostok.Game.join(roomId, "player-1");
}

// 取消匹配
Vostok.Game.cancelMatch(ticketId);

手动 Tick（关闭自动调度时）

Vostok.Game.init(new VKGameConfig().autoStartTicker(false));

// 业务侧手动驱动
scheduledExecutor.scheduleAtFixedRate(() -> {
    Vostok.Game.tickOnce();
}, 0, 50, TimeUnit.MILLISECONDS);
